import java.util.List;
import java.util.ArrayList;

public class Test01 {

public static void main(String[] args) {
Set> mm = PotencnaMnozica.potencna(Set.of(1, 2, 3));
System.out.println(preoblikuj(mm));
}

private static extends Comparable> List> preoblikuj(Set> mm) {
List> ll = new ArrayList<>();
for (Set m: mm) {
List l = new ArrayList<>(m);
l.sort(null);
ll.add(l);
}
ll.sort((p, q) -> primerjaj(p, q));
return ll;
}

private static extends Comparable> int primerjaj(List p, List q) {
if (p.isEmpty() && q.isEmpty()) {
return 0;
}
if (p.isEmpty() || q.isEmpty()) {
return (p.isEmpty() ? -1 : 1);
}
if (p.get(0).compareTo(q.get(0)) != 0) {
return p.get(0).compareTo(q.get(0));
}
return primerjaj(p.subList(1, p.size()), q.subList(1, q.size()));
}
}

public Cas(int h, int min) {
this.h = h;
this.min = min;
}

private int vrednost() {
return 60 * this.h + this.min;
}

@Override

return (this.vrednost() - drugi.vrednost());
}

@Override
