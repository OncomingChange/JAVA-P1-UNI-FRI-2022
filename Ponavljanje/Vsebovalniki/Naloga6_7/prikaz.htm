private int minuta;

this.minuta = min;
}

}

public boolean equals(Object obj) {
if (this == obj) {
return true;

if (!(obj instanceof Cas)) {
return false;
}
Cas drugi = (Cas) obj;

}

}

public int compareTo(Cas drugi) {

}
return (this.minuta - drugi.minuta);
}
